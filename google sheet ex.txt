var sheetName = "Votes";

function doGet(e) {
  var action = e && e.parameter ? e.parameter.action : "";
  
  if (action === "getResults") {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
    if (!sheet) {
      return ContentService.createTextOutput(JSON.stringify({ success: true, votes: [] }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    var data = sheet.getDataRange().getValues();
    var votes = [];
    // Start from row 1 to skip headers
    for (var i = 1; i < data.length; i++) {
        // Headers: ["Timestamp", "AdmissionNumber", "VoterName", "PresidentVote", "SecretaryVote", "TreasurerVote", "Status"]
        votes.push({
          "Timestamp": data[i][0] || "",
          "Admission Number": data[i][1] || "",
          "Voter Name": data[i][2] || "",
          "President Vote": data[i][3] || "",
          "Secretary Vote": data[i][4] || "",
          "Treasurer Vote": data[i][5] || ""
        });
    }
    
    return ContentService.createTextOutput(JSON.stringify({ success: true, votes: votes }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // Default active status response
  return ContentService.createTextOutput(JSON.stringify({ status: "active" }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    // Extract the FormData variables that the frontend script.js sends natively
    var data = e.parameter;
    
    // Automatically save vote if an admission number is provided
    if (data.admissionNo) {
      return saveVote(data);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: "Missing admission number" }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function saveVote(data) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) {
    sheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet(sheetName);
    sheet.appendRow(["Timestamp", "AdmissionNumber", "VoterName", "PresidentVote", "SecretaryVote", "TreasurerVote", "Status"]);
  }
  
  // Check if already voted
  var existingData = sheet.getDataRange().getValues();
  for (var i = 1; i < existingData.length; i++) {
    if (existingData[i][1] === data.admissionNo) {
      return ContentService.createTextOutput(JSON.stringify({ success: false, error: "Already voted!" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Append the row matching the exact keys sent by script.js!
  sheet.appendRow([
    data.timestamp || new Date().toISOString(),
    data.admissionNo,
    data.voterName,
    data.presidentName,   
    data.secretaryName,   
    data.treasurerName,   
    "Completed"
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Vote saved successfully!" }))
    .setMimeType(ContentService.MimeType.JSON);
}
